ostream& operator<<(ostream& os, __int128 x) {
    if(x == 0) {
        os << '0';
        return os;
    }
    if(x < 0) {
        os << '-';
        x = -x;
    }
    char buf[50];
    int idx = 0;
    while(x > 0) {
        buf[idx++] = '0' + (int)(x % 10);
        x /= 10;
    }
    while(idx--) {
        os << buf[idx];
    }
    return os;
}
